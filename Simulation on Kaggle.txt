# 3. Create the repaired topology (with absolute paths for forcefields)
ff_base = "/kaggle/input/datasets/joydebnath121/gromacs-2024-bin/gromacs_2024/share/gromacs/top/"
with open(f"{source_dir}/topol.top", 'r') as f:
    content = f.read()
fixed_content = content.replace('charmm27.ff/', f'{ff_base}charmm27.ff/')
with open('topol_final.top', 'w') as f:
    f.write(fixed_content)

# 4. Link the library dictionaries (.dat files)
!ln -sf {ff_base}/*.dat .

print("🚀 Running Grompp...")
!GMXLIB=. ./gmx grompp -f smd.mdp -c npt.gro -p topol_final.top -o smd.tpr -maxwarn 2
